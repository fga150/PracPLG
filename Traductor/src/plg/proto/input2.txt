program : ejemplo{


 consts{
       const natural dim = 3;
    }       
      
   tipos{
       tipo integer[3] array;  
     }  
     
    vars{
    	var array a;
    }
    
    subprograms{
     subprogram : asigna(array * a,integer num2){
       vars{
          var natural i;
       }
	   instructions{
	     i = 0;
	     while (i < dim)do
	      a[i] = i * num2;
	      i = i + 1;
	     endwhile
	   }  
  }
 }
  
 instructions{
  call asigna(a = a, num2 = 3);
 }
}
